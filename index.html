Employees' tax calculation

                    let (x) be -> [Annual Income].
                    let (y) be -> [Salary Bracket].

                    let (t) be -> [Tax Rate].

                    let (T) be -> [(t) / 100] => for conversion % to decimal.

                    ∴ f(x) = [[(x - y) * T ] + z].
                    ∴ f(x) / 12.
